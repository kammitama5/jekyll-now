- create rectangular pattern given ```*```, width and height

```
function rectangle(char, width, height)
{
  var arr = []
  var x = (char.repeat(width))
  //console.log(x)
  for (var i = 0; i < height; i++){
    var z = (x)
    arr.push(z)
  }
 
  return arr.join("\n")
}
```

- power without Math.pow

```
function power(x,y){
  var mult = 1;
  for (var i = 0; i < y; i++)
  {
    mult = mult * x
  }
  return mult;
}
```
