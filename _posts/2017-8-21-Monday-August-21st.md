```
import turtle
"""
draw a square in turtle
"""

def draw_square():
	window = turtle.Screen()
	window.bgcolor("green")
	
	brad = turtle.Turtle()
	
	brad.color("violet", "yellow")
	brad.begin_fill()
	
	brad.forward(100)
	brad.right(90)
	brad.forward(100)
	brad.right(90)
	brad.forward(100)
	brad.right(90)
	brad.forward(100)
	brad.end_fill()

	window.exitonclick()

draw_square()
```

function isAnagram(stringOne, stringTwo) {
    if ((stringOne == "") && (stringTwo == "")){
      return true
    }
    else{
    var a = stringOne.toLowerCase().replace(/ /g,'')
    var b = stringTwo.toLowerCase().replace(/ /g,'')
    var c = a.split('').sort().join('')
    var d = b.split('').sort().join('')
